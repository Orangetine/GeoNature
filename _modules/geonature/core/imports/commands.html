def fix_mappings():
    for label, values in [
        ("Synthese GeoNature", synthese_fieldmappings),
        ("Format DEE (champs 10 char)", dee_fieldmappings),
    ]:
        mapping = db.session.execute(sa.select(FieldMapping).filter_by(label=label)).scalar_one()
        mapping.values = values
        db.session.commit()
